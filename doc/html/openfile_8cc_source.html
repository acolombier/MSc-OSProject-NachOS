89 //----------------------------------------------------------------------
90 // OpenFile::ReadAt/WriteAt
91 // Read/write a portion of a file, starting at "position".
92 // Return the number of bytes actually written or read, but has
93 // no side effects (except that Write modifies the file, of course).
94 //
95 // There is no guarantee the request starts or ends on an even disk sector
96 // boundary; however the disk only knows how to read/write a whole disk
97 // sector at a time. Thus:
98 //
99 // For ReadAt:
100 // We read in all of the full or partial sectors that are part of the
101 // request, but we only copy the part we are interested in.
102 // For WriteAt:
103 // We must first read in any sectors that will be partially written,
104 // so that we don't overwrite the unmodified portion. We then copy
105 // in the data that will be modified, and write back all the full
106 // or partial sectors that are part of the request.
107 //
108 // "into" -- the buffer to contain the data to be read from disk
109 // "from" -- the buffer containing the data to be written to disk
110 // "numBytes" -- the number of bytes to transfer
111 // "position" -- the offset within the file of the first byte to be
112 // read/written
113 //----------------------------------------------------------------------
114 
115 int
116 OpenFile::ReadAt(char *into, int numBytes, int position)
117 {
118  int fileLength = hdr->FileLength();
119  int i, firstSector, lastSector, numSectors;
120  char *buf;
121 
122  if ((numBytes <= 0) || (position >= fileLength))
123  return 0; // check request
124  if ((position + numBytes) > fileLength)
125  numBytes = fileLength - position;
126  DEBUG('f', "Reading %d bytes at %d, from file of length %d.\n",
127  numBytes, position, fileLength);
128 
129  firstSector = divRoundDown(position, SectorSize);
130  lastSector = divRoundDown(position + numBytes - 1, SectorSize);
131  numSectors = 1 + lastSector - firstSector;
132 
133  // read in all the full and partial sectors that we need
134  buf = new char[numSectors * SectorSize];
135  for (i = firstSector; i <= lastSector; i++)
136  synchDisk->ReadSector(hdr->ByteToSector(i * SectorSize),
137  &buf[(i - firstSector) * SectorSize]);
138 
139  // copy the part we want
140  bcopy(&buf[position - (firstSector * SectorSize)], into, numBytes);
141  delete [] buf;
142  return numBytes;
143 }

145 int
146 OpenFile::WriteAt(const char *from, int numBytes, int position)
147 {
148  int fileLength = hdr->FileLength();
149  int i, firstSector, lastSector, numSectors;
150  bool firstAligned, lastAligned;
151  char *buf;
152 
153  if ((numBytes <= 0) || (position >= fileLength))
154  return 0; // check request
155  if ((position + numBytes) > fileLength)
156  numBytes = fileLength - position;
157  DEBUG('f', "Writing %d bytes at %d, from file of length %d.\n",
158  numBytes, position, fileLength);
159 
160  firstSector = divRoundDown(position, SectorSize);
161  lastSector = divRoundDown(position + numBytes - 1, SectorSize);
162  numSectors = 1 + lastSector - firstSector;
163 
164  buf = new char[numSectors * SectorSize];
165 
166  firstAligned = (position == (firstSector * SectorSize));
167  lastAligned = ((position + numBytes) == ((lastSector + 1) * SectorSize));
168 
169 // read in first and last sector, if they are to be partially modified
170  if (!firstAligned)
171  ReadAt(buf, SectorSize, firstSector * SectorSize);
172  if (!lastAligned && ((firstSector != lastSector) || firstAligned))
173  ReadAt(&buf[(lastSector - firstSector) * SectorSize],
174  SectorSize, lastSector * SectorSize);
175 
176 // copy in the bytes we want to change
177  bcopy(from, &buf[position - (firstSector * SectorSize)], numBytes);
178 
179 // write modified sectors back
180  for (i = firstSector; i <= lastSector; i++)
181  synchDisk->WriteSector(hdr->ByteToSector(i * SectorSize),
182  &buf[(i - firstSector) * SectorSize]);
183  delete [] buf;
184  return numBytes;
185 }
