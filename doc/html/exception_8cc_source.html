2 // Entry point into the Nachos kernel from user programs.
3 // There are two kinds of things that can cause control to
4 // transfer back to here from user code:
5 //
6 // syscall -- The user code explicitly requests to call a procedure
7 // in the Nachos kernel. Right now, the only function we support is
8 // "Halt".
9 //
10 // exceptions -- The user code does something that the CPU can't handle.
11 // For instance, accessing memory that doesn't exist, arithmetic errors,
12 // etc.
13 //
14 // Interrupts (which can also cause control to transfer from user
15 // code into the Nachos kernel) are handled elsewhere.
16 //
17 // For now, this only handles the Halt() system call.
18 // Everything else core dumps.
19 //

30 // UpdatePC : Increments the Program Counter register in order to resume
31 // the user program immediately after the "syscall" instruction.
32 //----------------------------------------------------------------------
33 static void
34 UpdatePC ()
35 {
36  int pc = machine->ReadRegister (PCReg);
37  machine->WriteRegister (PrevPCReg, pc);
38  pc = machine->ReadRegister (NextPCReg);
39  machine->WriteRegister (PCReg, pc);
40  pc += 4;
41  machine->WriteRegister (NextPCReg, pc);
42 }

46 // ExceptionHandler
47 // Entry point into the Nachos kernel. Called when a user program
48 // is executing, and either does a syscall, or generates an addressing
49 // or arithmetic exception.
50 //
51 // For system calls, the following is the calling convention:
52 //
53 // system call code -- r2
54 // arg1 -- r4
55 // arg2 -- r5
56 // arg3 -- r6
57 // arg4 -- r7
58 //
59 // The result of the system call, if any, must be put back into r2.
60 //
61 // And don't forget to increment the pc before returning. (Or else you'll
62 // loop making the same system call forever!
63 //
64 // "which" is the kind of exception. The list of possible exceptions
65 // are in machine.h.
66 //----------------------------------------------------------------------
67
