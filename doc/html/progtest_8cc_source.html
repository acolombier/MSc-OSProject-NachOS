78 void
79 ConsoleTest (char *in, char *out)
80 {
81  char ch;
82 
83  console = new Console (in, out, ReadAvail, WriteDone, 0);
84  readAvail = new Semaphore ("read avail", 0);
85  writeDone = new Semaphore ("write done", 0);
86 
87  for (;;)
88  {
89  readAvail->P (); // wait for character to arrive
90  ch = console->GetChar ();
91  console->PutChar (ch); // echo it!
92  writeDone->P (); // wait for write to finish
93  if (ch == 'q')
94  return; // if q, quit
95  }
96 }
