1 // main.cc
2 // Bootstrap code to initialize the operating system kernel.
3 //
4 // Allows direct calls into internal operating system functions,
5 // to simplify debugging and testing. In practice, the
6 // bootstrap code would just initialize data structures,
7 // and start a user program to print the login prompt.
8 //
9 // Most of this file is not needed until later assignments.
10 //
11 // Usage: nachos -d <debugflags> -rs <random seed #>
12 // -s -x <nachos file> -c <consoleIn> <consoleOut>
13 // -f -cp <unix file> <nachos file>
14 // -p <nachos file> -r <nachos file> -l -D -t
15 // -n <network reliability> -m <machine id>
16 // -o <other machine id>
17 // -z
18 //
19 // -d causes certain debugging messages to be printed (cf. utility.h)
20 // -rs causes Yield to occur at random (but repeatable) spots
21 // -z prints the copyright message
22 //
23 // USER_PROGRAM
24 // -s causes user programs to be executed in single-step mode
25 // -x runs a user program
26 // -c tests the console
27 // -sc tests the synchronous console
28 //
29 // FILESYS
30 // -f causes the physical disk to be formatted
31 // -cp copies a file from UNIX to Nachos
32 // -p prints a Nachos file to stdout
33 // -r removes a Nachos file from the file system
34 // -l lists the contents of the Nachos directory
35 // -D prints the contents of the entire file system
36 // -t tests the performance of the Nachos file system
37 //
38 // NETWORK
39 // -n sets the network reliability
40 // -m sets this machine's host id (needed for the network)
41 // -o runs a simple test of the Nachos network software
42 //
43 // NOTE -- flags are ignored until the relevant assignment.
44 // Some of the flags are interpreted here; some in system.cc.
45 //
46 // Copyright (c) 1992-1993 The Regents of the University of California.
47 // All rights reserved. See copyright.h for copyright notice and limitation
48 // of liability and disclaimer of warranty provisions.

67 // Bootstrap the operating system kernel.
68 //
69 // Check command line arguments
70 // Initialize data structures
71 // (optionally) Call test procedure
72 //
73 // "argc" is the number of command line arguments (including the name
74 // of the command) -- ex: "nachos -d +" -> argc = 3
75 // "argv" is an array of strings, one for each command line argument
76 // ex: "nachos -d +" -> argv = {"nachos", "-d", "+"}
77 //----------------------------------------------------------------------

79 int
80 main (int argc, char **argv)
81 {
82  int argCount; // the number of arguments
83  // for a particular command
84 
85  DEBUG ('t', "Entering main");
86  (void) Initialize (argc, argv);
87 
88 #ifdef THREADS
89  ThreadTest ();
90 #endif
91 
92  for (argc--, argv++; argc > 0; argc -= argCount, argv += argCount)
93  {
94  argCount = 1;
95  if (!strcmp (*argv, "-z")) // print copyright
96  printf ("%s", copyright);
97 #ifdef USER_PROGRAM
98  if (!strcmp (*argv, "-x"))
99  { // run a user program
100  ASSERT (argc > 1);
101  StartProcess (*(argv + 1));
102  argCount = 2;
103  }
104  else if (!strcmp (*argv, "-c"))
105  { // test the console
106  if (argc == 1)
107  ConsoleTest (NULL, NULL);
108  else
109  {
110  ASSERT (argc > 2);
111  ConsoleTest (*(argv + 1), *(argv + 2));
112  argCount = 3;
113  }
114  interrupt->Halt (); // once we start the console, then
115  // Nachos will loop forever waiting
116  // for console input
117  }
118  else if (!strcmp (*argv, "-sc"))
119  { // test the console
120  if (argc == 1)
121  SynchConsoleTest (NULL, NULL);
122  else
123  {
124  ASSERT (argc > 2);
125  SynchConsoleTest (*(argv + 1), *(argv + 2));
126  argCount = 3;
127  }
128  interrupt->Halt (); // once we start the console, then
129  // Nachos will loop forever waiting
130  // for console input
131  }
132 #endif // USER_PROGRAM
133 #ifdef FILESYS
134  if (!strcmp (*argv, "-cp"))
135  { // copy from UNIX to Nachos
136  ASSERT (argc > 2);
137  Copy (*(argv + 1), *(argv + 2));
138  argCount = 3;
139  }
140  else if (!strcmp (*argv, "-p"))
141  { // print a Nachos file
142  ASSERT (argc > 1);
143  Print (*(argv + 1));
144  argCount = 2;
145  }
146  else if (!strcmp (*argv, "-r"))
147  { // remove Nachos file
148  ASSERT (argc > 1);
149  fileSystem->Remove (*(argv + 1));
150  argCount = 2;
151  }
152  else if (!strcmp (*argv, "-l"))
153  { // list Nachos directory
154  fileSystem->List ();
155  }
156  else if (!strcmp (*argv, "-D"))
157  { // print entire filesystem
158  fileSystem->Print ();
159  }
160  else if (!strcmp (*argv, "-t"))
161  { // performance test
162  PerformanceTest ();
163  }
164 #endif // FILESYS
165 #ifdef NETWORK
166  if (!strcmp (*argv, "-o"))
167  {
168  ASSERT (argc > 1);
169  Delay (2); // delay for 2 seconds
170  // to give the user time to
171  // start up another nachos
172  MailTest (atoi (*(argv + 1)));
173  argCount = 2;
174  }
175 #endif // NETWORK
176  }
177 
178  currentThread->Finish (); // NOTE: if the procedure "main"
179  // returns, then the program "nachos"
180  // will exit (as any other normal program
181  // would). But there may be other
182  // threads on the ready list. We switch
183  // to those threads by saying that the
184  // "main" thread is finished, preventing
185  // it from returning.
186  return (0); // Not reached...
187 }
