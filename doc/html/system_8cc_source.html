45 // TimerInterruptHandler
46 // Interrupt handler for the timer device. The timer device is
47 // set up to interrupt the CPU periodically (once every TimerTicks).
48 // This routine is called each time there is a timer interrupt,
49 // with interrupts disabled.
50 //
51 // Note that instead of calling Yield() directly (which would
52 // suspend the interrupt handler, not the interrupted thread
53 // which is what we wanted to context switch), we set a flag
54 // so that once the interrupt handler is done, it will appear as
55 // if the interrupted thread called Yield at the point it is
56 // was interrupted.
57 //
58 // "dummy" is because every interrupt handler takes one argument,
59 // whether it needs it or not.
60 //----------------------------------------------------------------------
61 static void
62 TimerInterruptHandler (int dummy)
63 {
64  if (interrupt->getStatus () != IdleMode)
65  interrupt->YieldOnReturn ();
66 }

69 // Initialize
70 // Initialize Nachos global data structures. Interpret command
71 // line arguments in order to determine flags for the initialization.
72 //
73 // "argc" is the number of command line arguments (including the name
74 // of the command) -- ex: "nachos -d +" -> argc = 3
75 // "argv" is an array of strings, one for each command line argument
76 // ex: "nachos -d +" -> argv = {"nachos", "-d", "+"}
77 //----------------------------------------------------------------------
78 void
79 Initialize (int argc, char **argv)
80 {
81  int argCount;
82  const char *debugArgs = "";
83  bool randomYield = FALSE;
84 
85 #ifdef USER_PROGRAM
86  bool debugUserProg = FALSE; // single step user program
87 #endif
88 #ifdef FILESYS_NEEDED
89  bool format = FALSE; // format disk
90 #endif
91 #ifdef NETWORK
92  double rely = 1; // network reliability
93  int netname = 0; // UNIX socket name
94 #endif
95 
96  for (argc--, argv++; argc > 0; argc -= argCount, argv += argCount)
97  {
98  argCount = 1;
99  if (!strcmp (*argv, "-d"))
100  {
101  if (argc == 1)
102  debugArgs = "+"; // turn on all debug flags
103  else
104  {
105  debugArgs = *(argv + 1);
106  argCount = 2;
107  }
108  }
109  else if (!strcmp (*argv, "-rs"))
110  {
111  ASSERT (argc > 1);
112  RandomInit (atoi (*(argv + 1))); // initialize pseudo-random
113  // number generator
114  randomYield = TRUE;
115  argCount = 2;
116  }
117 #ifdef USER_PROGRAM
118  if (!strcmp (*argv, "-s"))
119  debugUserProg = TRUE;
120 #endif
121 #ifdef FILESYS_NEEDED
122  if (!strcmp (*argv, "-f"))
123  format = TRUE;
124 #endif
125 #ifdef NETWORK
126  if (!strcmp (*argv, "-l"))
127  {
128  ASSERT (argc > 1);
129  rely = atof (*(argv + 1));
130  argCount = 2;
131  }
132  else if (!strcmp (*argv, "-m"))
133  {
134  ASSERT (argc > 1);
135  netname = atoi (*(argv + 1));
136  argCount = 2;
137  }
138 #endif
139  }
140 
141  DebugInit (debugArgs); // initialize DEBUG messages
142  stats = new Statistics (); // collect statistics
143  interrupt = new Interrupt; // start up interrupt handling
144  scheduler = new Scheduler (); // initialize the ready queue
145  if (randomYield) // start the timer (if needed)
146  timer = new Timer (TimerInterruptHandler, 0, randomYield);
147 
148  threadToBeDestroyed = NULL;
149 
150  // We didn't explicitly allocate the current thread we are running in.
151  // But if it ever tries to give up the CPU, we better have a Thread
152  // object to save its state.
153  currentThread = new Thread ("main");
154  currentThread->setStatus (RUNNING);
155 
156  interrupt->Enable ();
157  CallOnUserAbort (Cleanup); // if user hits ctl-C
158 
159 #ifdef USER_PROGRAM
160  machine = new Machine (debugUserProg); // this must come first
161  synchconsole = new SynchConsole(NULL, NULL);
162 #endif
163 
164 #ifdef FILESYS
165  synchDisk = new SynchDisk ("DISK");
166 #endif
167 
168 #ifdef FILESYS_NEEDED
169  fileSystem = new FileSystem (format);
170 #endif
171 
172 #ifdef NETWORK
173  postOffice = new PostOffice (netname, rely, 10);
174 #endif
175 }
