3 // Routines to manage a singly-linked list of "things".
4 //
5 // A "ListElement" is allocated for each item to be put on the
6 // list; it is de-allocated when the item is removed. This means
7 // we don't need to keep a "next" pointer in every object we
8 // want to put on a list.
9 //
10 // NOTE: Mutual exclusion must be provided by the caller.
11 // If you want a synchronized list, you must use the routines

171 // List::SortedInsert
172 // Insert an "item" into a list, so that the list elements are
173 // sorted in increasing order by "sortKey".
174 //
175 // Allocate a ListElement to keep track of the item.
176 // If the list is empty, then this will be the only element.
177 // Otherwise, walk through the list, one element at a time,
178 // to find where the new item should be placed.
179 //
180 // "item" is the thing to put on the list, it can be a pointer to
181 // anything.
182 // "sortKey" is the priority of the item.
183 //----------------------------------------------------------------------
184 
185 void
186 List::SortedInsert (void *item, long long sortKey)
187 {
188  ListElement *element = new ListElement (item, sortKey);
189  ListElement *ptr; // keep track
190 
191  if (IsEmpty ())
192  { // if list is empty, put
193  first = element;
194  last = element;
195  }
196  else if (sortKey < first->key)
197  {
198  // item goes on front of list
199  element->next = first;
200  first = element;
201  }
202  else
203  { // look for first elt in list bigger than item
204  for (ptr = first; ptr->next != NULL; ptr = ptr->next)
205  {
206  if (sortKey < ptr->next->key)
207  {
208  element->next = ptr->next;
209  ptr->next = element;
210  return;
211  }
212  }
213  last->next = element; // item goes at end of list
214  last = element;
215  }
216 }

218 //----------------------------------------------------------------------
219 // List::SortedRemove
220 // Remove the first "item" from the front of a sorted list.
221 //
222 // Returns:
223 // Pointer to removed item, NULL if nothing on the list.
224 // Sets *keyPtr to the priority value of the removed item
225 // (this is needed by interrupt.cc, for instance).
226 //
227 // "keyPtr" is a pointer to the location in which to store the
228 // priority of the removed item.
229 //----------------------------------------------------------------------
230 
231 void *
232 List::SortedRemove (long long *keyPtr)
233 {
234  ListElement *element = first;
235  void *thing;
236 
237  if (IsEmpty ())
238  return NULL;
239 
240  thing = first->item;
241  if (first == last)
242  { // list had one item, now has none
243  first = NULL;
244  last = NULL;
245  }
246  else
247  {
248  first = element->next;
249  }
250  if (keyPtr != NULL)
251  *keyPtr = element->key;
252  delete element;
253  return thing;
254 }
