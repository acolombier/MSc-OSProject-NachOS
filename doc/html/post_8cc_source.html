1 // post.cc
2 // Routines to deliver incoming network messages to the correct
3 // "address" -- a mailbox, or a holding area for incoming messages.
4 // This module operates just like the US postal service (in other
5 // words, it works, but it's slow, and you can't really be sure if
6 // your mail really got through!).
7 //
8 // Note that once we prepend the MailHdr to the outgoing message data,
9 // the combination (MailHdr plus data) looks like "data" to the Network
10 // device.
11 //
12 // The implementation synchronizes incoming messages with threads
13 // waiting for those messages.
14 //

110 // MailBox::Get
111 // Get a message from a mailbox, parsing it into the packet header,
112 // mailbox header, and data.
113 //
114 // The calling thread waits if there are no messages in the mailbox.
115 //
116 // "pktHdr" -- address to put: source, destination machine ID's
117 // "mailHdr" -- address to put: source, destination mailbox ID's
118 // "data" -- address to put: payload message data
119 //----------------------------------------------------------------------
120 
121 void
122 MailBox::Get(PacketHeader *pktHdr, MailHeader *mailHdr, char *data)
123 {
124  DEBUG('n', "Waiting for mail in mailbox\n");
125  Mail *mail = (Mail *) messages->Remove(); // remove message from list;
126  // will wait if list is empty
127 
128  *pktHdr = mail->pktHdr;
129  *mailHdr = mail->mailHdr;
130  if (DebugIsEnabled('n')) {
131  printf("Got mail from mailbox: ");
132  PrintHeader(*pktHdr, *mailHdr);
133  }
134  bcopy(mail->data, data, mail->mailHdr.length);
135  // copy the message data into
136  // the caller's buffer
137  delete mail; // we've copied out the stuff we
138  // need, we can now discard the message
139 }

157 //----------------------------------------------------------------------
158 // PostOffice::PostOffice
159 // Initialize a post office as a collection of mailboxes.
160 // Also initialize the network device, to allow post offices
161 // on different machines to deliver messages to one another.
162 //
163 // We use a separate thread "the postal worker" to wait for messages
164 // to arrive, and deliver them to the correct mailbox. Note that
165 // delivering messages to the mailboxes can't be done directly
166 // by the interrupt handlers, because it requires a Lock.
167 //
168 // "addr" is this machine's network ID
169 // "reliability" is the probability that a network packet will
170 // be delivered (e.g., reliability = 1 means the network never
171 // drops any packets; reliability = 0 means the network never
172 // delivers any packets)
173 // "nBoxes" is the number of mail boxes in this Post Office
174 //----------------------------------------------------------------------
175 
176 PostOffice::PostOffice(NetworkAddress addr, double reliability, int nBoxes)
177 {
178 // First, initialize the synchronization with the interrupt handlers
179  messageAvailable = new Semaphore("message available", 0);
180  messageSent = new Semaphore("message sent", 0);
181  sendLock = new Lock("message send lock");
182 
183 // Second, initialize the mailboxes
184  netAddr = addr;
185  numBoxes = nBoxes;
186  boxes = new MailBox[nBoxes];
187 
188 // Third, initialize the network; tell it which interrupt handlers to call
189  network = new Network(addr, reliability, ReadAvail, WriteDone, (int) this);
190 
191 
192 // Finally, create a thread whose sole job is to wait for incoming messages,
193 // and put them in the right mailbox.
194  Thread *t = new Thread("postal worker");
195 
196  t->Fork(PostalHelper, (int) this);
197 }

213 //----------------------------------------------------------------------
214 // PostOffice::PostalDelivery
215 // Wait for incoming messages, and put them in the right mailbox.
216 //
217 // Incoming messages have had the PacketHeader stripped off,
218 // but the MailHeader is still tacked on the front of the data.
219 //----------------------------------------------------------------------
220 
221 void
222 PostOffice::PostalDelivery()
223 {
224  PacketHeader pktHdr;
225  MailHeader mailHdr;
226  char *buffer = new char[MaxPacketSize];
227 
228  for (;;) {
229  // first, wait for a message
230  messageAvailable->P();
231  pktHdr = network->Receive(buffer);
232 
233  mailHdr = *(MailHeader *)buffer;
234  if (DebugIsEnabled('n')) {
235  printf("Putting mail into mailbox: ");
236  PrintHeader(pktHdr, mailHdr);
237  }
238 
239  // check that arriving message is legal!
240  ASSERT(0 <= mailHdr.to && mailHdr.to < numBoxes);
241  ASSERT(mailHdr.length <= MaxMailSize);
242 
243  // put into mailbox
244  boxes[mailHdr.to].Put(pktHdr, mailHdr, buffer + sizeof(MailHeader));
245  }
246 }

248 //----------------------------------------------------------------------
249 // PostOffice::Send
250 // Concatenate the MailHeader to the front of the data, and pass
251 // the result to the Network for delivery to the destination machine.
252 //
253 // Note that the MailHeader + data looks just like normal payload
254 // data to the Network.
255 //
256 // "pktHdr" -- source, destination machine ID's
257 // "mailHdr" -- source, destination mailbox ID's
258 // "data" -- payload message data
259 //----------------------------------------------------------------------
260 
261 void
262 PostOffice::Send(PacketHeader pktHdr, MailHeader mailHdr, const char* data)
263 {
264  char* buffer = new char[MaxPacketSize]; // space to hold concatenated
265  // mailHdr + data
266 
267  if (DebugIsEnabled('n')) {
268  printf("Post send: ");
269  PrintHeader(pktHdr, mailHdr);
270  }
271  ASSERT(mailHdr.length <= MaxMailSize);
272  ASSERT(0 <= mailHdr.to && mailHdr.to < numBoxes);
273 
274  // fill in pktHdr, for the Network layer
275  pktHdr.from = netAddr;
276  pktHdr.length = mailHdr.length + sizeof(MailHeader);
277 
278  // concatenate MailHeader and data
279  bcopy(&mailHdr, buffer, sizeof(MailHeader));
280  bcopy(data, buffer + sizeof(MailHeader), mailHdr.length);
281 
282  sendLock->Acquire(); // only one message can be sent
283  // to the network at any one time
284  network->Send(pktHdr, buffer);
285  messageSent->P(); // wait for interrupt to tell us
286  // ok to send the next message
287  sendLock->Release();
288 
289  delete [] buffer; // we've sent the message, so
290  // we can delete our buffer
291 }
