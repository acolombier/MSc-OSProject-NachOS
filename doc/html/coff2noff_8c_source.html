coff2noff.c
1 /* coff2noff.c
2  *
3  * This program reads in a COFF format file, and outputs a NOFF format file.
4  * The NOFF format is essentially just a simpler version of the COFF file,
5  * recording where each segment is in the NOFF file, and where it is to
6  * go in the virtual address space.
7  *
8  * Assumes coff file is linked with either
9  * gld with -N -Ttext 0
10  * ld with -N -T 0
11  * to make sure the object file has no shared text.
12  *
13  * Also assumes that the COFF file has at most 3 segments:
14  * .text -- read-only executable instructions
15  * .data -- initialized data
16  * .bss/.sbss -- uninitialized data (should be zero'd on program startup)
17  *
18  * Copyright (c) 1992-1993 The Regents of the University of California.
19  * All rights reserved. See copyright.h for copyright notice and limitation
20  * of liability and disclaimer of warranty provisions.
21  */

39 /* Routines for converting words and short words to and from the
40  * simulated machine's format of little endian. These end up
41  * being NOPs when the host machine is little endian.
42  */

44 unsigned int
45 WordToHost(unsigned int word) {
46 #ifdef HOST_IS_BIG_ENDIAN
47  register unsigned long result;
48  result = (word >> 24) & 0x000000ff;
49  result |= (word >> 8) & 0x0000ff00;
50  result |= (word << 8) & 0x00ff0000;
51  result |= (word << 24) & 0xff000000;
52  return result;
53 #else
54  return word;
55 #endif /* HOST_IS_BIG_ENDIAN */
56 }
57 
58 unsigned short
59 ShortToHost(unsigned short shortword) {
60 #if HOST_IS_BIG_ENDIAN
61  register unsigned short result;
62  result = (shortword << 8) & 0xff00;
63  result |= (shortword >> 8) & 0x00ff;
64  return result;
65 #else
66  return shortword;
67 #endif /* HOST_IS_BIG_ENDIAN */
68 }

108 /* open the COFF file (input) */
109  fdIn = open(argv[1], O_RDONLY, 0);
110  if (fdIn == -1) {
111  perror(argv[1]);
112  exit(1);
113  }
114 
115 /* open the NOFF file (output) */
116  fdOut = open(argv[2], O_WRONLY|O_CREAT|O_TRUNC , 0666);
117  if (fdIn == -1) {
118  perror(argv[2]);
119  exit(1);
120  }
121  noffFileName = argv[2];
122 
123 /* Read in the file header and check the magic number. */
124  ReadStruct(fdIn,fileh);
125  fileh.f_magic = ShortToHost(fileh.f_magic);
126  fileh.f_nscns = ShortToHost(fileh.f_nscns);
127  if (fileh.f_magic != MIPSELMAGIC) {
128  fprintf(stderr, "File is not a MIPSEL COFF file\n");
129  unlink(noffFileName);
130  exit(1);
131  }
132 
133 /* Read in the system header and check the magic number */
134  ReadStruct(fdIn,systemh);
135  systemh.magic = ShortToHost(systemh.magic);
136  if (systemh.magic != OMAGIC) {
137  fprintf(stderr, "File is not a OMAGIC file\n");
138  unlink(noffFileName);
139  exit(1);
140  }
141 
142 /* Read in the section headers. */
143  numsections = fileh.f_nscns;
144  printf("numsections %d \n",numsections);
145  sections = (struct scnhdr *)malloc(numsections * sizeof(struct scnhdr));
146  Read(fdIn, (char *) sections, numsections * sizeof(struct scnhdr));
147 
148  for (i = 0; i < numsections; i++) {
149  sections[i].s_paddr = WordToHost(sections[i].s_paddr);
150  sections[i].s_size = WordToHost(sections[i].s_size);
151  sections[i].s_scnptr = WordToHost(sections[i].s_scnptr);
152  }
153 
154  /* initialize the NOFF header, in case not all the segments are defined
155  * in the COFF file
156  */
157  noffH.noffMagic = NOFFMAGIC;
158  noffH.code.size = 0;
159  noffH.initData.size = 0;
160  noffH.uninitData.size = 0;
161 
162  /* Copy the segments in */
163  inNoffFile = sizeof(NoffHeader);
164  lseek(fdOut, inNoffFile, 0);
165  printf("Loading %d sections:\n", numsections);
166  for (i = 0; i < numsections; i++) {
167  printf("\t\"%.8s\", filepos 0x%lx, mempos 0x%lx, size 0x%lx, flags 0x%lx\n",
168  sections[i].s_name, sections[i].s_scnptr,
169  sections[i].s_paddr, sections[i].s_size,
170  sections[i].s_flags);
171  if (sections[i].s_size == 0) {
172  /* do nothing! */
173  } else if (!strncmp(sections[i].s_name, ".text", 8)) {
174  noffH.code.virtualAddr = sections[i].s_paddr;
175  noffH.code.inFileAddr = inNoffFile;
176  noffH.code.size = sections[i].s_size;
177  lseek(fdIn, sections[i].s_scnptr, 0);
178  buffer = malloc(sections[i].s_size);
179  Read(fdIn, buffer, sections[i].s_size);
180  Write(fdOut, buffer, sections[i].s_size);
181  free(buffer);
182  inNoffFile += sections[i].s_size;
183  } else if (!strncmp(sections[i].s_name, ".data", 8)
184  || !strcmp(sections[i].s_name, ".rdata")) {
185  /* need to check if we have both .data and .rdata
186  * -- make sure one or the other is empty! */
187  if (noffH.initData.size != 0) {
188  fprintf(stderr, "Can't handle both data and rdata\n");
189  unlink(noffFileName);
190  exit(1);
191  }
192  noffH.initData.virtualAddr = sections[i].s_paddr;
193  noffH.initData.inFileAddr = inNoffFile;
194  noffH.initData.size = sections[i].s_size;
195  lseek(fdIn, sections[i].s_scnptr, 0);
196  buffer = malloc(sections[i].s_size);
197  Read(fdIn, buffer, sections[i].s_size);
198  Write(fdOut, buffer, sections[i].s_size);
199  free(buffer);
200  inNoffFile += sections[i].s_size;
201  } else if (!strncmp(sections[i].s_name, ".bss", 8) ||
202  !strcmp(sections[i].s_name, ".sbss")) {
203  /* need to check if we have both .bss and .sbss -- make sure they
204  * are contiguous
205  */
206  if (noffH.uninitData.size != 0) {
207  if (sections[i].s_paddr == (noffH.uninitData.virtualAddr +
208  noffH.uninitData.size)) {
209  fprintf(stderr, "Can't handle both bss and sbss\n");
210  unlink(noffFileName);
211  exit(1);
212  }
213  noffH.uninitData.size += sections[i].s_size;
214  } else {
215  noffH.uninitData.virtualAddr = sections[i].s_paddr;
216  noffH.uninitData.size = sections[i].s_size;
217  }
218  /* we don't need to copy the uninitialized data! */
219  } else {
220  fprintf(stderr, "Unknown segment type: %.8s\n", sections[i].s_name);
221  printf("\tFound %.8s section with contents:\n", sections[i].s_name);
222  lseek(fdIn, sections[i].s_scnptr, 0);
223  buffer = malloc(sections[i].s_size);
224  Read(fdIn, buffer, sections[i].s_size);
225  char *car, *car2, *car_lim;
226  for(car=buffer; car<buffer+sections[i].s_size; car +=8) {
227  printf("\t\t");
228  car_lim=car+8;
229  if (car_lim > buffer+sections[i].s_size) {
230  car_lim = buffer+sections[i].s_size;
231  }
232  for(car2=car; car2<car_lim; car2++) {
233  printf("%02x ", *car2);
234  }
235  for(car2=car; car2<car_lim; car2++) {
236  if (0 <= *car2 && *car2 <=31) {
237  printf(".");
238  } else {
239  printf("%c", *car2);
240  }
241  }
242  printf("\n");
243  }
244  Write(fdOut, buffer, sections[i].s_size);
245  free(buffer);
246  unlink(noffFileName);
247  exit(1);
248  }
249  }
250  lseek(fdOut, 0, 0);
251  Write(fdOut, (char *)&noffH, sizeof(NoffHeader));
252  close(fdIn);
253  close(fdOut);
254  exit(0);
255 }
