56 // if a packet is already buffered, we simply delay reading
57 // the incoming packet. In real life, the incoming
58 // packet might be dropped if we can't read it in time.
59 void
60 Network::CheckPktAvail()
61 {
62  // schedule the next time to poll for a packet
63  interrupt->Schedule(NetworkReadPoll, (int)this, NetworkTime, NetworkRecvInt);
64 
65  if (inHdr.length != 0) // do nothing if packet is already buffered
66  return;
67  if (!PollSocket(sock)) // do nothing if no packet to be read
68  return;
69 
70  // otherwise, read packet in
71  char *buffer = new char[MaxWireSize];
72  ReadFromSocket(sock, buffer, MaxWireSize);
73 
74  // divide packet into header and data
75  inHdr = *(PacketHeader *)buffer;
76  ASSERT((inHdr.to == ident) && (inHdr.length <= MaxPacketSize));
77  bcopy(buffer + sizeof(PacketHeader), inbox, inHdr.length);
78  delete []buffer ;
79 
80  DEBUG('n', "Network received packet from %d, length %d...\n",
81  (int) inHdr.from, inHdr.length);
82  stats->numPacketsRecvd++;
83 
84  // tell post office that the packet has arrived
85  (*readHandler)(handlerArg);
86 }

97 // send a packet by concatenating hdr and data, and schedule
98 // an interrupt to tell the user when the next packet can be sent
99 //
100 // Note we always pad out a packet to MaxWireSize before putting it into
101 // the socket, because it's simpler at the receive end.
102 void
103 Network::Send(PacketHeader hdr, char* data)
104 {
105  char toName[32];
106 
107  sprintf(toName, "SOCKET_%d", (int)hdr.to);
108 
109  ASSERT((sendBusy == FALSE) && (hdr.length > 0)
110  && (hdr.length <= MaxPacketSize) && (hdr.from == ident));
111  DEBUG('n', "Sending to addr %d, %d bytes... ", hdr.to, hdr.length);
112 
113  interrupt->Schedule(NetworkSendDone, (int)this, NetworkTime, NetworkSendInt);
114 
115  if (Random() % 100 >= chanceToWork * 100) { // emulate a lost packet
116  DEBUG('n', "oops, lost it!\n");
117  return;
118  }
119 
120  // concatenate hdr and data into a single buffer, and send it out
121  char *buffer = new char[MaxWireSize];
122  *(PacketHeader *)buffer = hdr;
123  bcopy(data, buffer + sizeof(PacketHeader), hdr.length);
124  SendToSocket(sock, buffer, MaxWireSize, toName);
125  delete []buffer;
126 }
